### Modèles à concevoir (context_models.py)

```python
class PageContext(BaseModel):
    """Contexte d'une page individuelle (intent, page_type, URL)."""
    model_config = ConfigDict(extra='forbid', frozen=True)
    slug: str
    page_type: Literal["home", "lp_sales", "lp_listicle", "advertorial", "pdp", "pricing", "blog", ...]
    url: str
    intent: str  # extracted from client_intent.json
    language: str = "en"

class ClientContext(BaseModel):
    """Contexte client global (brand DNA, ICP, doctrine version)."""
    model_config = ConfigDict(extra='forbid', frozen=True)
    slug: str
    brand_name: str
    industry: str
    icp_description: str
    doctrine_version: Literal["v3.2.1", "v3.3"] = "v3.2.1"

class ContextPackInput(BaseModel):
    model_config = ConfigDict(extra='forbid', frozen=True)
    slug: str
    page_type: str
    options: dict[str, Any] = Field(default_factory=dict)

class ContextPackOutput(BaseModel):
    """Output public de context_pack — consommé par persona_narrator, mode_1, multi_judge."""
    model_config = ConfigDict(extra='forbid', frozen=True)
    page: PageContext
    client: ClientContext
    capture_meta: dict[str, Any] = Field(default_factory=dict)
    perception_summary: dict[str, Any] = Field(default_factory=dict)
    scoring_summary: dict[str, Any] = Field(default_factory=dict)
    # NOTE: dict[str, Any] escape hatches : à typer en sprint follow-up dédié
```
