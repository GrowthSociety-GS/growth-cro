### Modèles à concevoir (visual_models.py)

```python
# Signatures cibles (LOC budget ~200)
class VisualBlock(BaseModel):
    """Bloc visuel détecté (cluster spatial perception V13)."""
    model_config = ConfigDict(extra='forbid', frozen=True)
    block_id: str
    role: Literal["hero", "cta", "social_proof", "feature", "footer", "nav", "other"]
    bbox: tuple[int, int, int, int]
    text_content: str | None = None
    confidence: float

class VisualHierarchy(BaseModel):
    model_config = ConfigDict(extra='forbid', frozen=True)
    blocks: list[VisualBlock]
    primary_block_id: str
    reading_order: list[str]

class VisualScore(BaseModel):
    model_config = ConfigDict(extra='forbid', frozen=True)
    pillar: Literal["hero", "persuasion", "ux", "coherence", "psycho", "tech"]
    score: float  # 0-1 normalized
    raw_points: int  # absolute points
    notes: str | None = None

class VisualReport(BaseModel):
    """Output public de visual_intelligence — frontière inter-module."""
    model_config = ConfigDict(extra='forbid', frozen=True)
    slug: str
    page_type: str
    hierarchy: VisualHierarchy
    scores: list[VisualScore]
    metadata: dict[str, Any] = Field(default_factory=dict)  # escape hatch pour debug, à typer dans sprint follow-up
```
