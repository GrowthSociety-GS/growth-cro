### Architecture flow
```
Webapp UI (TriggerRunButton)
  ↓ POST /api/runs/capture { client_slug, page_type }
Next.js route handler
  ↓ requireAdmin() → Supabase insert into runs (status=pending)
  ↓ returns { run_id }
Webapp UI mounts <RunStatusPill runId={...} />
  ↓ Supabase Realtime channel public:runs filtered by run_id
  ↓ status updates: pending → running → completed/failed
Worker daemon (local machine Mathis)
  ↓ poll Supabase every 30s
  ↓ pickup pending row, update status=running
  ↓ execute `python -m growthcro.capture.cli --url ... --label ... --page-type ...`
  ↓ stream stdout to runs.metadata_json.log_tail
  ↓ on completion: status=completed, output_path filled
  ↓ Supabase Realtime emits update
Webapp UI <RunStatusPill /> updates live
```

### Worker daemon design
```python
# growthcro/worker/daemon.py (≤500 LOC mono-concern)
import time, subprocess, signal
from growthcro.config import config
from growthcro.observability.logger import get_logger
from growthcro.worker.dispatcher import dispatch_run, RunRow

logger = get_logger(__name__)

def main_loop(poll_interval: int = 30, once: bool = False) -> int:
    """Poll Supabase runs queue and dispatch pending rows."""
    running = True
    def shutdown(*_):
        nonlocal running; running = False
    signal.signal(signal.SIGINT, shutdown)
    while running:
        try:
            pending = fetch_pending_runs()  # GET /rest/v1/runs?status=eq.pending&limit=5
            for run in pending:
                claim_run(run.id)  # UPDATE status=running
                try:
                    output = dispatch_run(run)  # executes CLI per run.type
                    complete_run(run.id, output)
                except Exception as e:
                    fail_run(run.id, str(e))
            if once: break
            time.sleep(poll_interval)
        except Exception as e:
            logger.error("worker loop error", extra={"err": str(e)})
            time.sleep(poll_interval)
    return 0
```

### Dispatcher (separates by run.type)
```python
# growthcro/worker/dispatcher.py
RUN_TYPE_TO_CLI = {
    "capture": ["python", "-m", "growthcro.capture.cli"],
    "score":   ["python", "-m", "growthcro.scoring.cli"],
    "recos":   ["python", "-m", "growthcro.recos.cli", "enrich"],
    "gsg":     ["python", "-m", "moteur_gsg.orchestrator"],
    # ...
}
def dispatch_run(run: RunRow) -> dict:
    cmd = RUN_TYPE_TO_CLI[run.type] + build_args(run.metadata_json)
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=600)
    return {"stdout_tail": result.stdout[-2000:], "returncode": result.returncode}
```
