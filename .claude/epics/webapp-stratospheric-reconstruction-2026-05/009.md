### Presence score computation
```python
# growthcro/geo/scorer.py
def compute_presence_score(response: str, brand: str, brand_keywords: list[str]) -> float:
    score = 0.0
    if re.search(rf'\b{re.escape(brand)}\b', response, re.IGNORECASE):
        score += 0.5
    keyword_hits = sum(1 for kw in brand_keywords if kw.lower() in response.lower())
    score += min(0.5, keyword_hits / len(brand_keywords) * 0.5)
    return round(score, 3)
```
