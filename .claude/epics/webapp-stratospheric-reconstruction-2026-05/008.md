### Sample size formula (pure TS, no new dep)
```typescript
// sample-size-calc.ts
function inverseNormal(p: number): number {
  // Acklam approximation for inverse normal CDF
  // Returns z-value for given probability

export function calculateSampleSize(opts: {
  baselineCvr: number;  // 0-1
  mde: number;          // 0-1 (e.g. 0.1 for +10% relative)
  alpha?: number;       // default 0.05
  power?: number;       // default 0.8
  tailed?: 1 | 2;       // default 2
}): { perArm: number; total: number } {
  const { baselineCvr, mde, alpha = 0.05, power = 0.8, tailed = 2 } = opts;
  const p1 = baselineCvr;
  const p2 = baselineCvr * (1 + mde);
  const zAlpha = inverseNormal(1 - alpha / tailed);
  const zBeta = inverseNormal(power);
  const n = Math.ceil(
    Math.pow(zAlpha + zBeta, 2) *
    (p1 * (1 - p1) + p2 * (1 - p2)) /
    Math.pow(p1 - p2, 2)
  );
  return { perArm: n, total: n * 2 };
}
```
