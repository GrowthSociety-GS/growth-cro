### AuditStatusPill design
```tsx
function AuditStatusPill({ status }: { status: AuditStatus }) {
  const config = {
    idle:      { tone: 'soft',  label: 'idle',      pulse: false },
    capturing: { tone: 'cyan',  label: 'capturing', pulse: true  },
    scoring:   { tone: 'violet',label: 'scoring',   pulse: true  },
    enriching: { tone: 'gold',  label: 'enriching', pulse: true  },
    done:      { tone: 'green', label: 'done',      pulse: false },
    failed:    { tone: 'red',   label: 'failed',    pulse: false },
  }[status];
  return <Pill tone={config.tone} pulse={config.pulse}>{config.label}</Pill>;
}
```
